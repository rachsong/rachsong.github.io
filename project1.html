''' Rachel Song Project1 CS110 S26 Simple Nature Scene ''' # importing everything from the turtle package from turtle import * # setting the background color bgcolor("skyblue") # variables sun_color = "yellow" house_color = "brown" door_color = "black" size = 100 # draw the sun penup() goto(200, 150) pendown() fillcolor(sun_color) begin_fill() circle(40) end_fill() # draw the house penup() goto(-50, -100) pendown() fillcolor(house_color) begin_fill() for i in range(4): # loop for square forward(size) left(90) end_fill() # draw the roof penup() goto(-50, 0) pendown() setheading(0) fillcolor("red") begin_fill() for i in range(3): forward(100) left(120) end_fill() # draw the door using a conditional penup() goto(-10, -100) pendown() if door_color == "black": fillcolor("black") else: fillcolor("gray") begin_fill() for i in range(2): forward(20) left(90) forward(40) left(90) end_fill() # draw a tree # trunk penup() goto(100, -100) pendown() fillcolor("brown") begin_fill() for i in range(2): forward(20) left(90) forward(50) left(90) end_fill() # leaves (top circle) penup() goto(110, -50) pendown() fillcolor("green") begin_fill() circle(30) end_fill() # so that the window won't go away done()
