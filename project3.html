''' Rachel Song Project3 CS110 S26 This program is an updated version of my second project which involves turtle graphics scene. I have refined my code by breaking down the original draw_scene() function into several other helper functions including draw_ground(), draw_sun(), draw_cloud(), draw_house(), draw_tree(), and draw_pathway(). This process helped me make my code more readable and structured. I also used some other techniques like removing duplicate code through function refactoring to make the scene more populous by adding more clouds, trees, houses, windows with lines, and more efficient pathways. ''' import turtle import math import time def setup_turtle(): t = turtle.Turtle() t.speed(0) screen = turtle.Screen() screen.title("Turtle Graphics Assignment") return t, screen def draw_rectangle(t, width, height, fill_color=None): if fill_color: t.fillcolor(fill_color) t.begin_fill() for _ in range(2): t.forward(width) t.right(90) t.forward(height) t.right(90) if fill_color: t.end_fill() def draw_square(t, size, fill_color=None): if fill_color: t.fillcolor(fill_color) t.begin_fill() for _ in range(4): t.forward(size) t.right(90) if fill_color: t.end_fill() def draw_triangle(t, size, fill_color=None): if fill_color: t.fillcolor(fill_color) t.begin_fill() for _ in range(3): t.forward(size) t.left(120) if fill_color: t.end_fill() def draw_circle(t, radius, fill_color=None): if fill_color: t.fillcolor(fill_color) t.begin_fill() t.circle(radius) if fill_color: t.end_fill() def jump_to(t, x, y): t.penup() t.goto(x, y) t.pendown() def draw_ground(t, y=-100): jump_to(t, -400, y) draw_rectangle(t, 800, 400, "lightgreen") def draw_sun(t, x=250, y=200): jump_to(t, x, y) draw_circle(t, 40, "yellow") def draw_cloud(t, x, y, size=1): jump_to(t, x, y) draw_circle(t, 25 * size, "white") jump_to(t, x + 30 * size, y + 20 * size) draw_circle(t, 30 * size, "white") jump_to(t, x + 70 * size, y) draw_circle(t, 25 * size, "white") def draw_window(t, x, y): jump_to(t, x, y) draw_square(t, 30, "lightblue") # vertical line jump_to(t, x + 15, y) t.setheading(270) t.forward(30) # horizontal line jump_to(t, x, y - 15) t.setheading(0) t.forward(30) def draw_house(t, x, y, add_windows=False): jump_to(t, x, y) draw_square(t, 160, "tan") jump_to(t, x, y) draw_triangle(t, 160, "red") jump_to(t, x + 60, y - 100) draw_rectangle(t, 40, 60, "brown") jump_to(t, x + 95, y - 130) draw_circle(t, 5, "gold") if add_windows: draw_window(t, x + 20, y - 40) draw_window(t, x + 110, y - 40) def draw_tree(t, x, y): jump_to(t, x, y) draw_rectangle(t, 30, 80, "brown") jump_to(t, x - 20, y - 20) draw_circle(t, 30, "green") jump_to(t, x + 5, y) draw_circle(t, 35, "green") jump_to(t, x + 40, y - 20) draw_circle(t, 30, "green") def draw_pathway(t, x, y, top_width=40, bottom_width=90, height=220): jump_to(t, x, y) t.fillcolor("gray") t.begin_fill() t.setheading(0) t.forward(top_width) t.goto(x + bottom_width, y - height) t.goto(x - (bottom_width - top_width), y - height) t.goto(x, y) t.end_fill() def draw_original_scene(t): screen = t.getscreen() screen.bgcolor("skyblue") draw_ground(t, -100) draw_sun(t, 250, 200) draw_cloud(t, -240, 230, 1) draw_house(t, -80, 60, False) draw_tree(t, -250, -20) # pathway jump_to(t, -30, -100) draw_rectangle(t, 60, 300, "gray") def draw_enhanced_scene(t): screen = t.getscreen() screen.bgcolor("skyblue") draw_ground(t, -50) draw_sun(t, 250, 200) draw_cloud(t, -240, 230, 1) draw_cloud(t, 40, 240, 0.8) draw_cloud(t, -30, 170, 1) draw_house(t, -80, 60, True) draw_house(t, 150, 80, True) # trees draw_tree(t, -300, -20) draw_tree(t, -200, -10) draw_tree(t, 100, -10) draw_tree(t, 310, -10) # pathways draw_pathway(t, -20, -100, 40, 90, 300) draw_pathway(t, 210, -80, 40, 90, 250) def draw_scene(t): draw_original_scene(t) time.sleep(2) # fixed pause t.clear() draw_enhanced_scene(t) def main(): t, screen = setup_turtle() draw_scene(t) screen.mainloop() if __name__ == "__main__": main()
